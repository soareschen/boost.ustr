template<typename StringT, typename CharT = typename char_type<StringT>::type> 
class string_traits {
public:
  // types
  typedef string_traits< StringT >                       this_traits;                  
  typedef StringT                                        string_type;                  
  typedef CharT                                          codeunit_type;                
  typedef string_type *                                  raw_strptr_type;              
  typedef const string_type *                            const_raw_strptr_type;        
  typedef boost::shared_ptr< const string_type >         const_strptr_type;            
  typedef std::auto_ptr< string_type >                   mutable_strptr_type;          
  typedef string_type::const_iterator                    codeunit_iterator_type;       
  typedef std::back_insert_iterator< string_type >       codeunit_output_iterator_type;
  typedef util::get_raw_char_type< codeunit_type >::type raw_char_type;                
  typedef std::basic_string< raw_char_type >             raw_string_type;              

  // member classes/structs/unions

  struct const_strptr {

    // public static functions
    static const_raw_strptr_type get(const const_strptr_type &);
    static codeunit_iterator_type codeunit_begin(const const_strptr_type &);
    static void reset(const_strptr_type &, raw_strptr_type);
    static codeunit_iterator_type codeunit_end(const const_strptr_type &);
    static bool equals(const_strptr_type, const_strptr_type);
  };

  struct mutable_strptr {

    // public static functions
    static raw_strptr_type release(mutable_strptr_type &);
    static raw_strptr_type release(mutable_strptr_type &&);
    static raw_strptr_type get(mutable_strptr_type &);
    static void append(mutable_strptr_type &, codeunit_type);
    static codeunit_output_iterator_type 
    output_iterator(mutable_strptr_type &);
    static void check_and_initialize(mutable_strptr_type &);
  };

  struct raw_strptr {

    // public static functions
    static bool equals(raw_strptr_type, raw_strptr_type);
    static bool equals(raw_strptr_type, const string_type &);
    static void delete_string(raw_strptr_type);
  };

  struct string {

    // public static functions
    static bool equals(const string_type &, const string_type &);
    static string_type from_raw_string(const raw_string_type &);
    template<typename Iterator> 
      static string_type from_iter(Iterator, Iterator);
  };

  // public static functions
  static raw_strptr_type new_string(const string_type &);
  static raw_strptr_type new_string();
  static raw_strptr_type clone_string(raw_strptr_type);

  // public data members
  static const size_t codeunit_size;
  static const string_type empty_string;
};
